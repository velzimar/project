TRI à bulle 


Definition :

Le principe du tri bulle (bubble sort) est de comparer deux à deux les éléments e1 et e2 consécutifs d'un 
tableau et d'effecteur une permutation si e1 > e2. On continue de trier jusqu'à ce qu'il n'y ait plus de permutation.

Remarque :

- Un tableau d'enregistrement est trié selon l'un des champs de l'enregistrement.


- Pour trier un fichier en ramène ces éléments dans un tableau qui sera trier par la suite puis on recopie les éléments du 
tableau trié dans le fichier


Traduction en Pascal :

program tri_tab;
uses wincrt; 
type 
     tab= array [1..25] of integer; 
var 
    t:tab; 
    n:integer; 
 
procedure saisie(var n:integer; var t:tab); 
var 
    i:integer; 
begin 
 repeat 
  writeln('donnner le nbre d''element du tableau'); 
  read(n); 
 until n in [5..25];
 
 For i:=1 to n do 
  begin 
   writeln('donner l''element ',i); 
   read(t[i]); 
  end; 
end; 
 
procedure tri(n:integer; var t:tab); 
var 
    verif:boolean; 
    i,x:integer; 
begin 
 repeat 
  verif:=true; 
   for i := 1 to n-1 do 
    begin 
     if t[i]>t[i+1] then 
      begin 
       x:=t[i]; 
       t[i]:=t[i+1]; 
       t[i+1]:=x; 
       verif:=false; 
      end; 
    end; 
 until verif=true; 
end; 
 
procedure affiche(n:integer; t:tab); 
var 
    i:integer; 
begin 
 write('| '); 
  for i:=1 to n do 
   begin 
    write(t[i],' | '); 
   end; 
end; 
 
BEGIN 
 saisie(n,t); 
 tri(n,t); 
 affiche(n,t); 
END.



TRI par insertion 


Definition :

Le principe du tri par insertion est d'insérer à la n-ième itération le n-ième élément à la bonne place.

Traduction en Pascal :

uses wincrt ; 
type 
    tab=array[1..20] of integer ; 
var 
 n,x,i : integer ; t: tab ; 
 
procedure saisie (var n : integer ;var t:tab); 
var 
i : integer ; 
Begin 
 repeat 
  writeln('saisir la taille du tableau :'); 
  readln(n); 
 until n in [5..20]   ;  
  For i:= 1 to n do 
   begin 
    repeat 
     writeln('Saisir l''element ',i); 
     readln(t[i]); 
    until t[i] > 0 ; 
   end; 
End; 
 
function position (t:tab ; i : integer ): integer ; 
var 
   j : integer ; 
Begin 
 j:=0 ; 
  repeat 
   j:=j+1 ; 
  until t[j] >= t[i] ; 
 position:=j ; 
End; 
 
procedure tri(var t : tab ; n:integer ); 
var 
int,i,j,p : integer ; 
Begin 
 for i:=2 to n do 
  begin 
   p:=position(t,i); 
    if p <> i then 
     begin 
      int:=t[i] ; 
       for j :=i-1 downto p do 
        begin 
         t[j+1]:=t[j] ; 
        end; 
      t[p]:=int ; 
     end; 
  end;  
End; 
 
procedure affiche (var t :tab ; n : integer ); 
var 
   i:integer ; 
begin 
 for i:= 1 to n-1 do 
  begin 
   write(t[i]); 
   write('|'); 
  end; 
   write(t[n]); 
end; 
 
begin 
 saisie(n,t); 
 x:=position(t,i); 
 tri(t,n); 
 affiche(t,n); 
end.




TRI par sélection


Definition :

Le principe du tri par sélection consiste à : 
1- déterminer le plus petit élément de l'élément (ou le plus grand) 
2- le placer à sa position finale (c'est-à-dire en dernière ou en première position du tableau) 
3- rechercher le second plus grand élément (ou le second plus petit élément) 
4- le placer à sa position finale (c'est-à-dire en avant-dernière ou en seconde position) 
5- répéter ce traitement jusqu’à ce que le tableau soit trié.


Traduction en Pascal :

Program TriInsertion ;
uses wincrt ;
type
   tab=array [1..20] of integer ;    
var
   n,x,p : integer ; t : tab ;
 
procedure saisie (var n : integer ; var t :tab );   
var
   i : integer ;   
begin 
 repeat
  writeln('Saisir la taille du tableau :');
  readln(n);
 until n in [5..20] ;
  For i:= 1 to n do                     
   begin
    repeat
     writeln('Saisir l''élément',i) ;
     readln(t[i]) ;
    until (t[i] > 0)   
   end;  
end;
 
function pos_max(n : integer ;t :tab; p :integer):integer ; 
var
   max,i : integer ;
Begin
     max:=p ;
 for i:= p to n do
  begin
   if t[i]>t[max] then
    max:=i ;
  end;
   pos_max:=max ; 
End;
 
procedure trie (n : integer ; var t :tab )   ;
var
   i,max,p : integer ;   
Begin  
 for i:= 1 to n-1 do
  begin   
   p:=pos_max(n,t,i);
   x:=t[i];
   t[i]:=t[p];
   t[p]:=x ;
  end;
End;  
 
procedure affiche (n : integer ; t :tab) ;
var
   i : integer ; 
Begin
 for i:= 1 to n-1 do
  begin
   write(t[i]);
   write('|');
  end;
   write(t[n]); 
End;
 
Begin       
 saisie(n,t); 
 trie(n,t);
 affiche(n,t);
End.






TRI Shell


Definition :

* Le tri shell est une amélioration de tri par insertion, au lieu de faire une rotation entre tous les éléments on réalise une rotation 
entre des éléments éloignès de p position (p = pas)


* On commence par un pas élévé et on diminue à chaque fois le pas jusqu’à avoir un pas égale à 1.


* Le pas calculé en utilisant la formule de suite suivante: ( p0 = 1 ; pk = pk-1 + 1 )

Traduction en Pascal :


program Tri_Shell;
uses wincrt;
type
 tab = array[1..20] of integer ;
var
 t : tab ;
 n : integer ;

 Procedure Shell(var t : tab ; n : integer);
  var
   p,i,j,aux : integer ;
  begin
   p := 1;
   while (p ‹ n) do
    begin
     p := p*3 + 1 ;
    end;
   while ( p<>1 ) do
    begin 
     p := p div 3 ;
     for i := p+1 to n do
      begin
       if (t[i] ‹ t[i-p]) then
        begin
         aux := t[i];
         j := i;
         repeat
          t[j] := t[j-p] ;
          j := j-p ;
         until (j ‹= p) or (t[j-p] ‹ aux) ;
         t[j] := aux;
        end;
      end;
    end;
  end;

 Procedure saisir(var t : tab ; var n : integer);
  var
   i : integer ;
  begin
   repeat
    writeln('Donner N');
    readln(N);
   until (n in [2..20]);
   for i := 1 to n do
    readln(t[i]);
  end;

 Procedure affiche(t:tab; n : integer);
  var
   i : integer ;
  begin
   write('|');
   for i:= 1 to n do 
   write(t[i],'|');
 end;

begin
 Saisir(t,n);
 Shell(t,n);
 Affiche(t,n);
end.




TRI fusion


Definition :


Le tri fusion consiste à diviser le tableau en deux parties, à trier chacune d'elle 
puis à la fusionner pour trier une partie, on refait le même traitement déjà décrit 
on utilisant des apelles récursives ppur diviser le tableau.




Traduction en Pascal :


Program Tri_Fusion;
Uses wincrt;

type
    tab = array [1..50] of integer;
var
    t : tab ;
    n,k : integer ;
    Procedure saisir (var t : tab ; var n : integer ) ;
              var
                 i : integer ;
              begin
                   repeat
                         writeln('Donner la taille du tableau');
                         readln(n);
                   until ( n in [10..50] ) ;
                   For i:= 1 to n do
                       begin
                            
                            t[i] := random(99)+100; 
                       end;
              end;

    Procedure Fusion(var t : tab ; d,m,f : integer );
              var
                 a : tab ;
                 n,c,i,j : integer ;   
              begin
                   i := d ;
                   j := m + 1 ;
                   n := d - 1 ;
                   repeat
                         n := n + 1 ; 
                         if (t[i] < t[j]) then
                            begin
                                 a[n] := t[i] ;
                                 i := i + 1 ;
                            end
                         else
                             begin
                                  a[n] := t[j] ;
                                  j := j + 1 ; 
                             end;      
                   until (i > m) or (j > f);
                   {si}
                   if (i>m) then
                      for c := j to f do
                          begin
                               n := n + 1 ;
                               a[n] := t[c] ;
                          end
                   else if (j>f) then
                        begin
                             for c := i to m do
                                 begin
                                      n := n + 1 ;
                                      a[n] := t[c];
                                 end;
                        end;
                   {fin si}
                   for i:= d to f do
                       t[i] := a[i] ;
              end;

    Procedure Tri (var t : tab ; d,f : integer);
              var
                 m : integer ;
              begin
                   if (d < f ) then
                     begin
                      m:= (d+f) div 2 ;
                      tri(t,d,m);
                      tri(t,m+1,f);
                      fusion(t,d,m,f);
                     end;
              end;

    Procedure affiche(t : tab ; n : integer ) ;
              var
                 i : integer ;
              begin
                   {----------------------Affichage---------------------}
                   for i := 1 to n do
                       begin
                            write(t[i],' | ');
                       end;     
              end;
begin
     saisir(t,n);
     randomize;
     Tri(t,1,n);
     affiche(t,n);
end.



TRI rapide


Definition :


Le tri rapide consiste à choisir un élément appelé pivot et à placer ceux qui lui sont infèrieure à sa gauche et ceux qui lui sont 
supèrieure à sa droite.On recommence ce traitement sur la partie gauche et la partie droite du tableau.
L'élément pivot peut être le premier,le dèrnier ou le milieu.


Traduction en Pascal :


Program Tri_Rapide;
Uses wincrt;

type
    tab = array [1..50] of integer;
var
    t : tab ;
    n,k : integer ;
    Procedure saisir (var t : tab ; var n : integer ) ;
              var
                 i : integer ;
              begin
                   repeat
                         writeln('Donner la taille du tableau');
                         readln(n);
                   until ( n in [10..50] ) ;
                   For i:= 1 to n do
                       begin
                            
                            t[i] := random(99)+100; 
                       end;
              end;

    Procedure Pivot(var t : tab ; d,f : integer; var p : integer );
              var
                 aux,i : integer ;   
              begin
                   p := d ;
                   for i:= d+1 to f do
                       begin
                            if (t[i] < t[d] ) then
                               begin
                                    p:= p+1;
                                    aux := t[p] ;
                                    t[p] := t[i] ;
                                    t[i] := aux;
                               end;
                       end;
                   aux  := t[d] ;
                   t[d] := t[p] ;
                   t[p] := aux  ;
              end;

    Procedure rapide(var t : tab ; d,f : integer);
              var
                 p : integer ;
              begin
                   if (d < f ) then
                     begin
                      Pivot(t,d,f,p);
                      Rapide(t,d,p-1);
                      Rapide(t,p+1,f);
                     end;
              end;

    Procedure affiche(t : tab ; n : integer ) ;
              var
                 i : integer ;
              begin
                   {----------------------Affichage---------------------}
                   for i := 1 to n do
                       begin
                            write(t[i],' | ');
                       end;     
              end;
begin
     saisir(t,n);
     randomize;
     rapide(t,1,n);
     affiche(t,n);
end.
