Les Facteurs Premiers

Définition:

Décomposer un entier en facteur premier c'est à dire l'écrire sous la forme de produit de ces facteurs premier.

Traduction en Pascal:

 
program Facteur_Premier ;
uses wincrt ;
var n : longint ;
Function Fact_premier ( n:longint ): string ;
var 
	i : longint ;
	ch,ch2 : string ;
begin
	i := 2 ;
	ch := '' ;
	repeat 
		if ( n mod i ) = 0 then 
			begin
				str(i,ch2);
				ch := ch + ch2 + '*' ;
				n := n div i
			end	  
		else
			i := i + 1 ;	 
	until ( n = 1 );
	delete(ch,length(ch),1);
	Fact_premier := ch ;
end;
begin
	readln(n);
	write(Fact_premier(n));
end.
