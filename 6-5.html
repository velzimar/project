Définition:

Un entier supérieur à 1 est dit premier s’il n’est divisible que par 1 et par lui même.


function Premier ( n :integer ) : boolean ;
var i : integer ;
begin
	i:=2 ; 
	while (n mod i <> 0 ) and ( i <= n div 2 ) do
		begin
			i := i + 1 ;
		end;
	premier := (i > n div 2) ;
end;
