Définition:

Convertir un entier positif N dans une base b au moins égale à 2 consiste à chercher la représentation de N dans la base b.

Conversion base 10 en base 2:

 
0) Def Fn C_10_2_Iterative ( n:entier ) : chaine 
1) Repeter
		R := n mod 2
		convch(R,ch)
		Bin := ch + Bin
		n := n div 2
   Jusqu'à ( n = 0 )
2) C_10_2_Iterative := Bin
3) Fin C_10_2_Iterative   
	
0) Def Fn C_10_2_Recursive ( n:entier ) : chaine 
1) Si ( n = 0 ) alors
		C_10_2_Recursive := ""
	Sinon
		R := n mod 2
		convch(R,ch)
		C_10_2_Recursive := C_10_2_Recursive( n  div 2 ) + ch
	Fin si	
3) Fin C_10_2_Recursive

Conversion base 10 en base 16:

 
0) Def Fn C_10_16_Iterative ( n:entier ) : chaine
1) [Hex := ""]
	Repeter 
		R := n mod 16
		Si (R >= 10) alors
			ch := chr(R+55)
		Sinon
			convch(R,ch)
		Fin si
		Hex := ch + Hex
		n := n div 16
	Jusqu'à ( n = 0 )
2) C_10_16_Iterative := Hex	
3) Fin C_10_16_Iterative

Conversion base 16 en base 10:

 
0) Def Fn C_16_10_Iterative ( Hex:chaine ) : entier	
1)[p := 1 , x := 0] 
	Pour i do 1 à long(Hex) faire
		Si (Hex[i] dans ['A'..'F']) alors
			x := ord(Hex[i])-55
		Sinon 
			valeur(Hex[i],a,e)
		Fin si
		x := x + a * p
		p := p * 16
	Fin pour
2) C_16_10_Iterative := x
3) Fin C_16_10_Iterative

Conversion base 16 en base 2:

1)[Bin := ""] 
	Pour i de 1 à long(Hex) faire
		Si (Hex[i] dans ["A".."F"]) alors
			a := ord(Hex[i]) - 55
		Sinon
			Valeur(Hex[i],a,er)
		Fin si
		ch := C_10_2(a)
		Tant que (long(ch) < a) faire
			ch := "0" + ch
		Fin tant que
		Bin := Bin + ch
	Fin pour	
	Tant que ( Bin[1] = "0" ) faire
		effacer(bin,1,1)
	Fin tant que
2) C_16_2_Iterative := Bin
3) Fin C_16_2_Iterative
