PGCD

Le plus grand commun diviseur de deux entiers a et b appelé aussi PGCD est le plus grand entier permettant de diviser a et b. On le note PGCD (a,b). 


Function PGCD_DEUCLIDE_ITERATIVE (A,B:integer):integer;
var 
	r : integer ;
begin
	while ( (a mod b) <> 0 ) do
		begin
			r := a mod b ;
			a := b ;
			b := r ;
		end;
	PGCD_EUCLIDE_ITERATIVE := b ;
end;
Function PGCD_EUCLIDE_RECURSIVE (A,B:integer):integer;
begin
	If ( a mod b = 0) then
		PGCD_EUCLIDE_RECURSIVE := b 
	else
		PGCD_EUCLIDE_RECURSIVE := PGCD_EUCLIDE_RECURSIVE(b,a mod b);
end;


Function  PGCD_DIFFERENCE_ITERATIVE (A,B:integer):integer;
var 
	r : integer;
begin
	while (b <> a) do
		if (a > b)then
			a := a - b 
		else
			b := b - a ;
	PGCD_DIFFERENCE_ITERATIVE := a ;
end;
Function PGCD_DIFFERENCE_RECURSIVE (A,B:integer):integer;
begin 
	If (a=b) then
		PGCD_DIFFERENCE_RECURSIVE := a ;
	else if (a > b) then
		PGCD_DIFFERENCE_RECURSIVE := PGCD_DIFFERENCE_RECURSIVE(a-b,b)
	else
		PGCD_DIFFERENCE_RECURSIVE := PGCD_DIFFERENCE_RECURSIVE(a,b-a);
end;

PPCM

Le plus petit commun multiple de deux entiers a et b appelé aussi PPCM est le plus petit entier multiple à la fois de a et de b. On le note PPCM (a,b). 


Function PPCM_ITERATIVE ( a,b : integer): integer ;
var 
	max,min : integer ;
begin
	if a > b then 
		begin
			max := a ;
			min := b ;
		end
	else
		begin 
			max := b ; 
			min := a ;
		end;
	while ( max mod min <> 0 ) do 
		begin
			max := max + ( a + b - min );
		end ;
	PPCM_ITERATIVE := max ;
end ;
Function PPCM_RECURSIVE ( a,b,max,min : integer): integer ;
Begin
	If (max mod min = 0) then
		PPCM_RECURSIVE := max 
	else if (a>b) then
		PPCM_RECURSIVE := PPCM_RECURSIVE(a,b,max+a,min)
	else
		PPCM_RECURSIVE := PPCM_RECURSIVE(a,b,max+b,min);
end ;
